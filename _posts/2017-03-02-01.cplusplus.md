#### 1.1. printf와 scanf를 대신하는 입출력 방식
---
* 출력

```cpp
#include <iostream>

int main(void)
{
	int num=20;
	std::cout<<"Hello World!"<<std::endl;
	std::cout<<"Hello "<<"World!"<<std::endl;
	std::cout<<num<<' '<<'A';
	std::cout<<' '<<3.14<<std::endl;
	return 0;
}
```

* 출력 및 입력(정수)

```cpp
#include <iostream>

int main(void)
{
	int val1, val2;
	int result=0;
	std::cout<<"두 개의 숫자입력: ";
	std::cin>>val1>>val2;

	if(val1<val2)
	{
		for(int i=val1+1; i<val2; i++)
			result+=i;
	}
	else
	{
		for(int i=val2+1; i<val1; i++)
			result+=i;
	}

	std::cout<<"두 수 사이의 정수 합: "<<result<<std::endl;
	return 0;
}
```

* 출력 및 입력(문자열)

```cpp
#include <iostream>

int main(void)
{

	std::cout<<"제일 좋아하는 언어는 "<<lang<<"입니다."<<std::endl;

}
```

#### 1.2. 함수 오버로딩
---
* 매개변수의 자료형 또는 개수를 다르게하여 오버로딩

```
#include <iostream>

void MyFunc(void)
{
	std::cout<<"MyFunc(void) called"<<std::endl;
}

void MyFunc(char c)
{
	std::cout<<"MyFunc(char c) called"<<std::endl;
}

void MyFunc(int a, int b)
{
	std::cout<<"MyFunc(int a, int b) called"<<std::endl;
}

int main(void)
{
	MyFunc();
	MyFunc('A');
	MyFunc(12, 13);
	return 0;
}
```


#### 1.3. 매개변수의 디폴트 값
---
* 함수의 선언부가 없는경우

```cpp
#include <iostream>

int Adder(int num1=1, int num2=2)
{
	return num1+num2;
}

int main(void)
{
	std::cout<<Adder()<<std::endl;
	std::cout<<Adder(5)<<std::endl;
	std::cout<<Adder(3, 5)<<std::endl;
	return 0;
}
```

* 함수의 선언부가 있는경우

```cpp
#include <iostream>

int Adder(int num1=1, int num2=2);

int main(void)
{
	std::cout<<Adder()<<std::endl;
	std::cout<<Adder(5)<<std::endl;
	std::cout<<Adder(3, 5)<<std::endl;
	return 0;
}

int Adder(int num1, int num2)
{
	return num1+num2;
}
```

* 부분적으로 디폴트값을 적용하는 경우(우->좌)

#include <iostream>

* alias(범위지정 연산자의 depth가 있는경우)

```
#include <iostream>
using namespace std;

namespace AAA
{
	namespace BBB
	{
		namespace CCC
		{
			int num1;
			int num2;
		}
	}
}

int main(void)
{
	AAA::BBB::CCC::num1=20;
	AAA::BBB::CCC::num2=30;

	namespace ABC=AAA::BBB::CCC;

	cout<<ABC::num1<<endl;
	cout<<ABC::num2<<endl;
	return 0;
}
```
